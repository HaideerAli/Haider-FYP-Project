7
Dockerize & Deploy
Write Dockerfile and docker-compose for backend + frontend. Deploy backend on Render/Railway, frontend on Vercel. Configure environment variables and production builds.

§ Data Preprocessing

Transform raw medical CSV data into a clean, ML-ready format. This phase is critical — model accuracy depends on this step.

VSCode Prompt — Full Preprocessing Pipeline
📋 VSCode AI Prompt
Create `src/preprocessing/preprocess.py` with a complete preprocessing pipeline for a medicine recommendation system. Include: 1. Load dataset from data/raw/ using pandas 2. Clean: strip whitespace from symptom columns, fill NaN symptoms with empty string 3. Encode symptoms using sklearn MultiLabelBinarizer — create binary symptom matrix (0/1 for each symptom) 4. Encode target disease column with LabelEncoder 5. Handle class imbalance using SMOTE from imbalanced-learn 6. Split data 80/20 train/test with stratify=y and random_state=42 7. Save processed data: X_train, X_test, y_train, y_test as numpy arrays in data/processed/ 8. Save encoders: symptom_encoder.pkl, label_encoder.pkl, symptom_list.json using joblib 9. Print full preprocessing report with shapes, class distribution, and symptom count 10. Add error handling and logging throughout Use: pandas, numpy, sklearn, imbalanced-learn, joblib
⌨
Core Preprocessing Code
import pandas as pd import numpy as np from sklearn.preprocessing import LabelEncoder from sklearn.model_selection import train_test_split from sklearn.preprocessing import MultiLabelBinarizer from imblearn.over_sampling import SMOTE import joblib, json, os def preprocess(csv_path: str): df = pd.read_csv(csv_path) # Get symptom columns (all except 'Disease') symptom_cols = [c for c in df.columns if c != 'Disease'] # Collect symptoms per row as list df['symptoms_list'] = df[symptom_cols].apply( lambda row: [s.strip() for s in row if pd.notna(s) and s != ''], axis=1 ) # MultiLabelBinarizer for symptoms mlb = MultiLabelBinarizer() X = mlb.fit_transform(df['symptoms_list']) # Label encode diseases le = LabelEncoder() y = le.fit_transform(df['Disease']) # SMOTE for class balance sm = SMOTE(random_state=42) X_res, y_res = sm.fit_resample(X, y) X_train, X_test, y_train, y_test = train_test_split( X_res, y_res, test_size=0.2, stratify=y_res, random_state=42 ) # Save artifacts joblib.dump(mlb, 'models/symptom_encoder.pkl') joblib.dump(le, 'models/label_encoder.pkl') json.dump(list(mlb.classes_), open('models/symptom_list.json', 'w')) return X_train, X_test, y_train, y_test, mlb, le
ℹ SMOTE (Synthetic Minority Over-sampling Technique) is essential here because medical datasets are highly imbalanced — some diseases appear much more than others. Without it, your model will predict common diseases for everything.

Model Training Core
from sklearn.ensemble import RandomForestClassifier from sklearn.svm import SVC from sklearn.naive_bayes import GaussianNB from sklearn.neural_network import MLPClassifier from sklearn.model_selection import GridSearchCV from xgboost import XGBClassifier import joblib, time, json MODELS = { "Random Forest": RandomForestClassifier(n_estimators=200, random_state=42), "XGBoost": XGBClassifier(use_label_encoder=False, eval_metric='mlogloss'), "SVM": SVC(kernel='rbf', probability=True), "Naive Bayes": GaussianNB(), "MLP": MLPClassifier(hidden_layer_sizes=(256,128,64), max_iter=500), } results = {} for name, model in MODELS.items(): t = time.time() model.fit(X_train, y_train) acc = model.score(X_test, y_test) results[name] = {"accuracy": acc, "time": round(time.time()-t, 2)} print(f"✓ {name}: {acc:.4f}") # Save best model best = max(results, key=lambda k: results[k]["accuracy"]) joblib.dump(MODELS[best], 'models/best_model.pkl') print(f"\n🏆 Best: {best} saved to models/best_model.pkl")

FastAPI Core — main.py
from fastapi import FastAPI, HTTPException from fastapi.middleware.cors import CORSMiddleware from pydantic import BaseModel, validator from typing import List import joblib, json, numpy as np app = FastAPI(title="MediRec API", version="1.0.0") app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"]) # Load models at startup model = symptom_enc = label_enc = symptom_list = None @app.on_event("startup") async def load_models(): global model, symptom_enc, label_enc, symptom_list model = joblib.load("../models/best_model.pkl") symptom_enc = joblib.load("../models/symptom_encoder.pkl") label_enc = joblib.load("../models/label_encoder.pkl") symptom_list = json.load(open("../models/symptom_list.json")) class PredictionRequest(BaseModel): symptoms: List[str] @validator('symptoms') def check_symptoms(cls, v): if not v: raise ValueError("Provide at least 1 symptom") return v @app.post("/api/predict") async def predict(req: PredictionRequest): encoded = symptom_enc.transform([req.symptoms]) proba = model.predict_proba(encoded)[0] top3_idx = np.argsort(proba)[::-1][:3] disease = label_enc.inverse_transform([top3_idx[0]])[0] return { "predicted_disease": disease, "confidence": round(float(proba[top3_idx[0]]) * 100, 2), "top_3": [label_enc.classes_[i] for i in top3_idx], "medicines": get_medicines(disease), "precautions": get_precautions(disease) }
